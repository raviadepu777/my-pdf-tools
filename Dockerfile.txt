# ==> STAGE 1: Build the application and install dependencies <==
FROM node:18-slim as builder

# Set the working directory for the build stage
WORKDIR /usr/src/app

# Copy package.json and package-lock.json (if it exists)
COPY package*.json ./

# Install all dependencies
RUN npm install

# ==> STAGE 2: Create the final, lean production image <==
FROM node:18-slim

# Install LibreOffice
RUN apt-get update && apt-get install -y \
    libreoffice \
    --no-install-recommends \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Set the working directory for the final image
WORKDIR /usr/src/app

# Copy the pre-built node_modules folder from the 'builder' stage
COPY --from=builder /usr/src/app/node_modules ./node_modules

# Copy the application source code
COPY . .

# Expose the port our app will run on
EXPOSE 3000

# This command is just a fallback; Render will override it.
CMD ["node", "server.js"]
